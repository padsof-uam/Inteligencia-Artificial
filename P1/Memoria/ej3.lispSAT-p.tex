\begin{aibox}{\function}
\begin{alltt}
;; SAT-p

SYNTAX: SAT-p (kb)
RECIBE   : base de conocimiento (KB)
EVALÚA A : T si existe al menos un modelo para kb
            NIL en caso contrario
\end{alltt}
\end{aibox}

\begin{aibox}{\examples}
\begin{alltt}
(SAT-p '((<=> A (¬ H)) (<=> P (^ A H)) (<=> H P))) 
; T

(SAT-p '((=> (^ P I) L) (=> (¬ P) (¬ L)) (¬ P) L)) 
; NIL
\end{alltt}

\end{aibox}

\begin{aibox}{\comments}

\end{aibox}
\begin{aibox}{\pseudocode}
\begin{alltt}
mods = encontrar modelos de kb
si mods es vacía
    devolver NIL
si no
    devolver T
\end{alltt}
\end{aibox}
\begin{aibox}{\code}

\begin{alltt}

(defun SAT-p (kb)
  (not (null (encuentra-modelos kb))))

\end{alltt}
\end{aibox}
